403 void Foam::populationBalanceModels::univariatePopulationBalance::solve()
404 {
405  surfaceScalarField phiOwn("phiOwn", fvc::interpolate(U_) & U_.mesh().Sf());
406  surfaceScalarField phiNei("phiNei", phiOwn);
407  updateAdvection(phiOwn, phiNei);
408 
409  // Integrate source and diffusion terms
410  forAll(quadrature_.moments(), mI)
411  {
412  volUnivariateMoment& m = quadrature_.moments()[mI];
413 
414  fvScalarMatrix momentEqn
415  (
416  fvm::ddt(m)
417  + advectMoment(m, phiOwn, phiNei)
418  - diffusionModel_->momentDiff(m)
419  ==
420  aggregationSource(m)
421  + breakupSource(m)
422  + growthSource(m)
423  );
424 
425  momentEqn.relax();
426  momentEqn.solve();
427 
428  }
429 
430  quadrature_.updateQuadrature();
431 }
